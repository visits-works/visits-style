import { Container } from '@components';
import Cube from '_components/Cube';
import theme from '@theme/default';

プロジェクト事に色を設定してください。


## public theme

<Container>
  {Object.keys(theme).map(key => (typeof theme[key] === 'string' && theme[key].indexOf('#') === 0) ? (
    <Cube key={key} color={theme[key]}>{key}<br />{theme[key]}</Cube>
  ) : null)}
</Container>
