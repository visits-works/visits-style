    <div
      key={key}

        width: '150px', height: '150px',
        textAlign: 'right', fontWeight: 600,
        padding: '0.5rem',
        margin: '0.5rem',
        color: findColorInvert(publicTheme[key]),

      {publicTheme[key]}

  ))}
